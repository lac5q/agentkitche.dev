```python
# State management pattern — MUST use identical atomic write
STATE_FILE = VAULT / "obsidian-ingestion-state.json"

def load_state() -> dict:
    if STATE_FILE.exists():
        with open(STATE_FILE) as f:
            return json.load(f)
    return {}

def save_state(state: dict):
    tmp = str(STATE_FILE) + ".tmp"
    with open(tmp, "w") as f:
        json.dump(state, f, indent=2, default=str)
    os.replace(tmp, str(STATE_FILE))  # atomic

def content_hash(text: str) -> str:
    return hashlib.sha256(text.encode("utf-8")).hexdigest()[:16]

def post_to_mem0(text: str, metadata: dict) -> bool:
    payload = json.dumps({
        "text": text,
        "agent_id": AGENT_ID,  # <-- projects uses "shared" (not "claude")
        "metadata": metadata,
    }).encode("utf-8")
    req = urllib.request.Request(
        f"{MEM0_URL}/memory/add",
        data=payload,
        headers={"Content-Type": "application/json"},
        method="POST",
    )
    try:
        with urllib.request.urlopen(req, timeout=120) as resp:
            return resp.status in (200, 201)
    except (urllib.error.URLError, OSError, TimeoutError) as e:
        print(f"  [WARN] mem0 unreachable: {e}", file=sys.stderr)
        return False
```

    ```python
    def main():
        if not PROJECTS_DIR.exists():
            print(f"[ERROR] Projects dir not found: {PROJECTS_DIR}", file=sys.stderr)
            sys.exit(1)

        state = load_state()
        total_ingested = 0
        total_skipped_hash = 0
        total_skipped_mtime = 0

        project_dirs = sorted(d for d in PROJECTS_DIR.iterdir() if d.is_dir())

        for project_dir in project_dirs:
            dirname = project_dir.name
            proj_state = state.setdefault(f"project-{dirname}", {
                "last_mtime": None,
                "processed_hashes": [],
            })

            last_mtime = proj_state.get("last_mtime")
            processed_hashes = set(proj_state.get("processed_hashes", []))
            cutoff_ts = datetime.fromisoformat(last_mtime) if last_mtime else None

            md_files = sorted(project_dir.rglob("*.md"))
            ingested = 0
            skipped_hash = 0
            skipped_mtime = 0

            for md_file in md_files:
                content = md_file.read_text(encoding="utf-8", errors="replace").strip()
                if not content:
                    continue

                h = content_hash(content)
                if h in processed_hashes:
                    skipped_hash += 1
                    continue

                mtime = datetime.fromtimestamp(md_file.stat().st_mtime)
                if cutoff_ts and mtime <= cutoff_ts:
                    skipped_mtime += 1
                    continue

                rel_path = md_file.relative_to(VAULT)
                print(f"  -> [{dirname}] {rel_path} ({len(content)} chars)")

                text = f"Project knowledge [{dirname}] {md_file.stem}:\n\n{content}"
                metadata = {
                    "source": SOURCE_TAG,
                    "project": dirname,
                    "file": str(md_file),
                    "hash": h,
                }

                if post_to_mem0(text, metadata):
                    processed_hashes.add(h)
                    ingested += 1
                    if proj_state["last_mtime"] is None or mtime.isoformat() > proj_state["last_mtime"]:
                        proj_state["last_mtime"] = mtime.isoformat()
                else:
                    print(f"  [WARN] Failed to sync {rel_path}, will retry next run")

            proj_state["processed_hashes"] = list(processed_hashes)
            total_ingested += ingested
            total_skipped_hash += skipped_hash
            total_skipped_mtime += skipped_mtime

        save_state(state)
        print(f"  Projects → mem0: {total_ingested} synced, {total_skipped_hash} duplicate, {total_skipped_mtime} unchanged")

    if __name__ == "__main__":
        main()
    ```
